  storeSessionToken(sessionToken) {
    return store.save(this.SESSION_TOKEN_KEY,{
      sessionToken: sessionToken
    });

  }

getSessionToken

Params
sessionToken Object the currentUser object from Parse.com When Hot Loading, the sessionToken will be passed in, and if so, it needs to be stored on the device. Remember, the store is a promise so, have to be careful. 
  getSessionToken(sessionToken) {
    if (sessionToken) {
      return store.save(this.SESSION_TOKEN_KEY,{
          sessionToken: sessionToken
      }).then(() => {
        return store.get(this.SESSION_TOKEN_KEY);
      });
    }
    return store.get(this.SESSION_TOKEN_KEY);
  }

deleteSessionToken

Deleted during log out

 
  deleteSessionToken() {
    return store.delete(this.SESSION_TOKEN_KEY);
  }
}
